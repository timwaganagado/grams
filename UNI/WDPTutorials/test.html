packages = ["numpy", "matplotlib"] import numpy as np import matplotlib.pyplot as plt x = [] y = [] # Generate 200 values between 0 and 2pi values = np.linspace(0, 2*np.pi, 100) for a in values: r = np.cos(20*a) x.append(r * np.cos(a)) y.append(r * np.sin(a)) plt.plot(x, y, c='b') plt.title('r = cos(20*a)') # Show the plot display(plt, target="output", append=False)
